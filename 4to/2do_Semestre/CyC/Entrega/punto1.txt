# 11101011010100

# 111011011110110100
(q0,0)=>(qf,0,D)

# 11101110111101110100
(q0,1)=>(qf,1,D)

#---------------------------qf------------------------------#
# El estado qf va al final de la cinta.                     #
#-----------------------------------------------------------#
# 1111011011110110100
(qf,0)=>(qf,0,D)

# 111101110111101110100
(qf,1)=>(qf,1,D)

# 1111010111110101100
(qf,B)=>(qc,B,I)

#--------------------------qc------------------------------#
# El estado qc inicia el corrimiento a izquierda           #
#----------------------------------------------------------#
# 1111101101111110101100
(qc,0)=>(qc0,B,I)

# 111110111011111110101100
(qc,1)=>(qc1,B,I)

#--------------------------qc0-----------------------------#
# El estado qc0 escribe un 0 en la la celda actual, y mueve#

#----------------------------------------------------------#

# 111110111011111101101100
(qc0,1)=>(qc1,0,I)

# 11111010101011100
(qc0,B)=>(qa,B,S)

#----------------------------------------------------------#

# 11111101110111111011101100
(qc1,1)=>(qc1,1,I)
