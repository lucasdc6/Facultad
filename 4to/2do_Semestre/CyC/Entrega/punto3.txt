# Definicion MT codificada
# Estados
# qa = 1
# qr = 11
# q0 = 111
# qi = 1^(1+3)

# Simbolos
# B  = 1
# a1 = 11
# a2 = 111
# ai = 1^(i+1)

# Movimientos
# D = 1
# I = 11
# S = 111

#-----------------------------------------------------------#
# El estado q0 es el estado en el que se verifica el input, #
# y mueve hasta la siguiente componente (2da)               #
#-----------------------------------------------------------#

# 1110101101011100
(q0,B)=>(qr,B,S)

# 111011011110110100
(q0,0)=>(q1,0,D)

# 1110111011101110100
(q0,1)=>(q0,1,D)

#-----------------------------------------------------------#
# El estado q1 es el que se detiene en la 3er componente    #
#-----------------------------------------------------------#

# 11110101101011100
(q1,B)=>(qr,B,S)

# 1111011010110100
(q1,0)=>(qa,0,D)

# 111101110111101110100
(q1,1)=>(q1,1,D)

#-----------------------------------------------------------#
# Codificacion completa                                     #
#-----------------------------------------------------------#
