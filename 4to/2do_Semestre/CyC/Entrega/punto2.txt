#-----------------------------------------------------------#
# El estado q0 es el estado en el que se verifica el input. #
#-----------------------------------------------------------#

# 1110101101011100
(q0,B)=>(qr,B,S)

# 1110110111101110100
(q0,0)=>(qd0,1,D)

# 111011101111101110100
(q0,1)=>(qd1,1,D)

#-----------------------------------------------------------#
# El estado qd0 es el estado que escribe un 0 en la celda   #
# actual, y mueve a la derecha                              #
#-----------------------------------------------------------#

# 111101010110100
(qd0,B)=>(qa,0,D)

# 1111011011110110100
(qd0,0)=>(qd0,0,D)

# 111101110111110110100
(qd0,1)=>(qd1,0,D)

#-----------------------------------------------------------#
# El estado qd1 es el estado que escribe un 1 en la celda   #
# actual, y mueve a la derecha                              #
#-----------------------------------------------------------#

# 11111010101110100
(qd1,B)=>(qa,1,D)

# 111110110111101110100
(qd1,0)=>(qd0,1,D)

# 11111011101111101110100
(qd1,1)=>(qd1,1,D)
